```php
// GCD (–ù–û–î) –∏ LCM (–ù–û–ö)
function gcd($a, $b) {
    return $b === 0 ? $a : gcd($b, $a % $b);
}

function lcm($a, $b) {
    return abs($a * $b) / gcd($a, $b);
}

// –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø—Ä–æ—Å—Ç–æ–µ —á–∏—Å–ª–æ
function isPrime($n) {
    if ($n < 2) return false;
    if ($n === 2) return true;
    if ($n % 2 === 0) return false;
    
    for ($i = 3; $i <= sqrt($n); $i += 2) {
        if ($n % $i === 0) return false;
    }
    return true;
}

// –§–∞–∫—Ç–æ—Ä–∏–∞–ª
function factorial($n) {
    return $n <= 1 ? 1 : $n * factorial($n - 1);
}

// –ß–∏—Å–ª–∞ –§–∏–±–æ–Ω–∞—á—á–∏
function fibonacci($n) {
    if ($n <= 1) return $n;
    $a = 0; $b = 1;
    for ($i = 2; $i <= $n; $i++) {
        [$a, $b] = [$b, $a + $b];
    }
    return $b;
}
```

```php
// Two Sum —Å HashMap
function twoSum($nums, $target) {
    $map = [];
    foreach ($nums as $i => $num) {
        $complement = $target - $num;
        if (isset($map[$complement])) {
            return [$map[$complement], $i];
        }
        $map[$num] = $i;
    }
    return [];
}

// –ü–æ–∏—Å–∫ –¥—É–±–ª–∏–∫–∞—Ç–æ–≤
function containsDuplicate($nums) {
    $seen = [];
    foreach ($nums as $num) {
        if (isset($seen[$num])) return true;
        $seen[$num] = true;
    }
    return false;
}

// Longest Substring Without Repeating Characters
function lengthOfLongestSubstring($s) {
    $charSet = [];
    $left = 0;
    $maxLen = 0;
    
    for ($right = 0; $right < strlen($s); $right++) {
        while (isset($charSet[$s[$right]])) {
            unset($charSet[$s[$left]]);
            $left++;
        }
        $charSet[$s[$right]] = true;
        $maxLen = max($maxLen, $right - $left + 1);
    }
    
    return $maxLen;
}
```

```php
// –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–∞–ª–∏–Ω–¥—Ä–æ–º–∞
function isPalindrome($s) {
    $left = 0;
    $right = strlen($s) - 1;
    
    while ($left < $right) {
        if ($s[$left] !== $s[$right]) {
            return false;
        }
        $left++;
        $right--;
    }
    return true;
}

// –£–¥–∞–ª–µ–Ω–∏–µ –¥—É–±–ª–∏–∫–∞—Ç–æ–≤ –∏–∑ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –º–∞—Å—Å–∏–≤–∞
function removeDuplicates(&$nums) {
    if (empty($nums)) return 0;
    
    $slow = 0;
    for ($fast = 1; $fast < count($nums); $fast++) {
        if ($nums[$fast] !== $nums[$slow]) {
            $slow++;
            $nums[$slow] = $nums[$fast];
        }
    }
    return $slow + 1;
}
```

### Sliding Window

```php

function maxSumSubarray($arr, $k) {
    $n = count($arr);
    if ($n < $k) return -1;
    
    // –ü–µ—Ä–≤–æ–µ –æ–∫–Ω–æ
    $windowSum = array_sum(array_slice($arr, 0, $k));
    $maxSum = $windowSum;
    
    // –°–¥–≤–∏–≥–∞–µ–º –æ–∫–Ω–æ
    for ($i = $k; $i < $n; $i++) {
        $windowSum += $arr[$i] - $arr[$i - $k];
        $maxSum = max($maxSum, $windowSum);
    }
    
    return $maxSum;
}
```
